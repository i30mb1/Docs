# Основные настройки сервера
listen              1935;
max_connections     20;
daemon              off;
srs_log_tank        console;

http_api {
    enabled         on;
    listen          1985;
}

http_server {
    enabled         on;
    listen          8080;
}

# Настройки SRT сервера (Прием потока с камеры)
srt_server {
    enabled         on;
    listen          10080;

    # Синхронизация задержки с камерой (1 секунда)
    latency         1000;

    # Плавность: отдает пакеты в OBS равномерно, убирая рывки
    tsbpdmode       on;

    # Защита: сбрасывает слишком старые пакеты, не давая задержке расти
    tlpktdrop       on;

    # Увеличенные буферы для работы с HEVC/1080p60
    sendbuf         20000000;
    recvbuf         20000000;

    connect_timeout 4000;
    peer_idle_timeout 8000;
}

# Виртуальный хост (Конвертация SRT -> RTMP для OBS)
vhost __defaultVhost__ {
    srt {
        enabled     on;
        srt_to_rtmp on;
    }
}
