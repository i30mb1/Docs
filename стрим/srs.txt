vhost __defaultVhost__ {
}

srt_server {
    enabled         on;
    listen          10080;
    
    # Синхронизируем с камерой
    latency         1000;      
    
    # ОБЯЗАТЕЛЬНО ВКЛЮЧИТЬ: Timestamp-based Packet Delivery
    # В режиме 'on' SRT отдает пакеты равномерно, как их видит камера.
    # Это уберет рывки и фризы в OBS.
    tsbpdmode       on;        
    
    # ОБЯЗАТЕЛЬНО ВКЛЮЧИТЬ: Too-late Packet Drop
    # Если пакет опоздал более чем на 1 секунду, он сбрасывается.
    # Это не дает задержке расти (стрим не будет отставать от чата всё сильнее).
    tlpktdrop       on;        
    
    # Большие буферы для 4K/1080p60
    sendbuf         20000000;
    recvbuf         20000000;
    
    connect_timeout 4000;
    peer_idle_timeout 8000;
    maxbw           -1;
    mss             1500;
}
